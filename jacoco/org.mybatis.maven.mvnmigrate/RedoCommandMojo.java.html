package org.mybatis.maven.mvnmigrate;

import org.apache.ibatis.migration.commands.RedoCommand;
import org.apache.ibatis.migration.options.SelectedOptions;
import org.apache.maven.plugin.MojoExecutionException;
import org.apache.maven.plugin.MojoFailureException;
import org.apache.maven.plugins.annotations.Mojo;
import org.apache.maven.plugins.annotations.Parameter;

/**
 * Goal which execute the ibatis migration redo command.
 */
@Mojo(name = "redo")
public final class RedoCommandMojo extends AbstractCommandMojo<RedoCommand> {

  /**
   * Steps to redo (default one step).
   */
  @Parameter(property = "migration.redo.steps")
  private String redoSteps;

  @Override
  protected RedoCommand createCommandClass(SelectedOptions options) {
    return new RedoCommand(options);
  }

  @Override
  public void execute() throws MojoExecutionException, MojoFailureException {
    if (isSkip()) {
      return;
    }

    init();
    getCommand().execute(redoSteps);
  }

}
